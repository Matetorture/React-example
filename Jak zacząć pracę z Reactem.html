Czym jest React?

React jest wydajną i elastyczną biblioteką JavaScript służącą do budowania złożonych interfejsów użytkownika przy użyciu małych i odizolowanych od siebie kawałków kodu, czyli komponentów.

Dlaczego warto korzystać z Reacta?

ZaletyWady

Jednokierunkowy przepływ danychWysoka częstotliwość aktualizacji bibliotek
Open-source
Wirtualny DOM
Komponenty wielokrotnego użytku
Możliwość renderowania po stronie serwera (SSR)
Duża społeczność deweloperów

Co będzie potrzebne by zacząć pracę z Reactem?

  • Node.js
  • Dowolny edytor kodu (polecamy Visual Studio Code)
  • Znajomość języków: HTML, CSS, JavaScript lub TypeScript

Stworzenie pierwszego projektu

Z użyciem npx

npx create-react-app my-app

Z użyciem npm

npm init react-app my-app

Z użyciem Yarn

yarn create react-app my-app

Włączenie projektu

cd my-app

npm start

Projekt będzie możliwy do zobaczenia:

http://localhost:3000
Link do poradnika React: https://react.dev/learn

Komponenty

komponenty można pisać za pomocą funkcji lub za pomocą klasy

Function - komponent za pomocą funkcji MyComponent.js

function MyComponent() {
  return (
    <div>
      <h1>Mój komponent</h1>
      <p>To jest przykładowy komponent funkcyjny.</p>
    </div>
  );
}

export default MyComponent;

Class - komponent za pomocą klasy MyComponent.js

import React from 'react';

class MyComponent extends React.Component {
  render() {
    return (
      <div>
        <h1>Mój komponent</h1>
        <p>To jest przykładowy komponent klasowy.</p>
      </div>
    );
  }
}

export default MyComponent;

Wykorzystanie komponentu w App.js 

import './App.css';
import MyComponent from './MyComponent';

function App() {
  return (
    <div className="App">
      <MyComponent/>
    </div>
  );
}

export default App;

State

służy do przechowywania i zarządzania dynamicznymi danymi specyficznymi dla danego komponentu. Jest to lokalny magazyn danych, który umożliwia komponentowi zapamiętywanie i aktualizowanie informacji bez wpływu na inne komponenty.

Counter.js

import React from 'react';

class Counter extends React.Component {
  constructor(props) {
    super(props);
    this.state = { count: 0 };
  }

  increment = () => {
    this.setState((prevState) => ({ count: prevState.count + 1 }));
  };

  render() {
    return (
      <div>
        <h1>Licznik: {this.state.count}</h1>
        <button onClick={this.increment}>Zwiększ</button>
      </div>
    );
  }
}

export default Counter;
import React, { useState } from 'react';

function Form() {
  const [count, setCount] = useState(0);

  const increment = () => {
    setCount((prevState) => prevState + 1);
  };

  return (
    <div>
      <h1>Licznik: {count}</h1>
      <button onClick={increment}>Zwiększ</button>
    </div>
  );
}

export default Form;

Przykład formularza

repo z projektem: https://github.com/Matetorture/React-example/tree/form/my-app

Form.js

import React, { useState } from 'react';

function Form() {
  const [name, setName] = useState({ firstName: 'Jan', lastName: 'K' });

  const handleChange = (event) => {
    setName({ ...name, [event.target.name]: event.target.value });
  };

  return (
    <div>
      <form>
        <label htmlFor="firstName">First name</label>
        <input
          type="text"
          name="firstName"
          value={name.firstName}
          onChange={handleChange}
        />

        <br/>
        <br/>

        <label htmlFor="lastName">Last name</label>
        <input
          type="text"
          name="lastName"
          value={name.lastName}
          onChange={handleChange}
        />
      </form>
      
      <br/>

      <span>{name.firstName} {name.lastName}</span>
    </div>
  );
}

export default Form;

Walidacja

repo z projektem: https://github.com/Matetorture/React-example/tree/formWithValidation/my-app

Form.js

import './Form.css';
import React, { useState } from 'react';

function Form() {
  const [input, setInput] = useState({ firstName: '', lastName: '' });
  const [errors, setErrors] = useState({ firstName: '', lastName: '' });

  const handleChange = (event) => {
    const { name, value } = event.target;
    setInput({ ...input, [name]: value });

    const newErrors = { ...errors };
    switch (name) {
      case 'firstName':
        if (value.length < 3) {
          newErrors.firstName = 'Imię musi mieć co najmniej 3 znaki';
        } else if (value.length > 64) {
          newErrors.firstName = 'Imię może mieć maksymalnie 64 znaki';
        } else if (/[\d\W_]/.test(value)) {
          newErrors.firstName = 'Imię nie może zawierać cyfr ani znaków specjalnych';
        } else {
          newErrors.firstName = '';
        }
        break;
      case 'lastName':
        if (value.length < 3) {
          newErrors.lastName = 'Nazwisko musi mieć co najmniej 3 znaki';
        } else if (value.length > 64) {
          newErrors.lastName = 'Nazwisko może mieć maksymalnie 64 znaki';
        } else if (/[\d\W_]/.test(value)) {
          newErrors.lastName = 'Nazwisko nie może zawierać cyfr ani znaków specjalnych';
        } else {
          newErrors.lastName = '';
        }
        break;
      default:
        break;
    }
    setErrors(newErrors);
  };

  const handleSubmit = (event) => {
    event.preventDefault();

    if (errors.firstName || errors.lastName || input.firstName === '' || input.lastName === '') {
      return;
    }

    console.log('Dane przesłane:', input);
  };

  return (
    <div>
      <form onSubmit={handleSubmit}>
        <label htmlFor="firstName">Imię</label>
        <input
          type="text"
          name="firstName"
          value={input.firstName}
          onChange={handleChange}
        />
        <br/>
        {errors.firstName && <span className="error">{errors.firstName}</span>}

        <br />
        <br />

        <label htmlFor="lastName">Nazwisko</label>
        <input
          type="text"
          name="lastName"
          value={input.lastName}
          onChange={handleChange}
        />
        <br/>
        {errors.lastName && <span className="error">{errors.lastName}</span>}

        <br />
        <br />

        <button type="submit">Wyślij</button>
      </form>
    </div>
  );
}

export default Form;
